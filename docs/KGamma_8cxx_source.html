382  cout << "event number: " << evtNo << endl;

385  // * Load event information and track collection *
386  /*
387  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
388  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)

391  SmartDataPtr<EvtRecEvent> evtRecEvent (eventSvc(), EventModel::EvtRec::EvtRecEvent);
392  SmartDataPtr<EvtRecTrackCol> evtRecTrkCol(eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
393 
394  // * Log number of events *
395  log << MSG::DEBUG << "Ncharged, Nneutral, Ntotal = "
396  << evtRecEvent->totalCharged() << ", "
397  << evtRecEvent->totalNeutral() << ", "
398  << evtRecEvent->totalTracks() << endreq;
399 
400 
401  // * Set vertex origin * //
402  Hep3Vector xorigin(0,0,0);
403  IVertexDbSvc* vtxsvc;
404  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
405  if(vtxsvc->isVertexValid()){
406  double* dbv = vtxsvc->PrimaryVertex();
407  double* vv = vtxsvc->SigmaPrimaryVertex();
408  // HepVector dbv = m_reader.PrimaryVertex(runNo);
409  // HepVector vv = m_reader.SigmaPrimaryVertex(runNo);
410  xorigin.setX(dbv[0]);
411  xorigin.setY(dbv[1]);
412  xorigin.setZ(dbv[2]);
413  }
414 
415 
416  // * LOOP OVER CHARGED TRACKS: select charged tracks (eventual pions) * //
417  // NOTE: Ncut1 -- nGood has to be 2, mdcTrk->charge() has to be more than 0
418  // The first part of the set of reconstructed tracks are the charged tracks
419  int nCharge = 0;
421  Vint iGood;
423  for(int i = 0; i < evtRecEvent->totalCharged(); ++i) {
424 
425  // * Get track: beginning of all tracks + event number
426  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
427  if(!(*itTrk)->isMdcTrackValid()) continue;
428 
429  // * Get track info from Main Drift Chamber
430  RecMdcTrack *mdcTrk = (*itTrk)->mdcTrack();
431  double pch = mdcTrk->p();
432  double x0 = mdcTrk->x();
433  double y0 = mdcTrk->y();
434  double z0 = mdcTrk->z();
435  double phi0 = mdcTrk->helix(1);
436 
437  // * Get vertex origin
438  double xv = xorigin.x();
439  double yv = xorigin.y();
440  double Rxy = (x0-xv)*cos(phi0)+(y0-yv)*sin(phi0);
441 
442  // * Get radii of vertex
443  HepVector a = mdcTrk->helix();
444  HepSymMatrix Ea = mdcTrk->err();
445  HepPoint3D point0(0.,0.,0.); // the initial point for MDC recosntruction
446  HepPoint3D IP(xorigin[0],xorigin[1],xorigin[2]);
447  VFHelix helixip(point0,a,Ea);
448  helixip.pivot(IP);
449  HepVector vecipa = helixip.a();
450  double Rvxy0 = fabs(vecipa[0]); // nearest distance to IP in xy plane
451  double Rvz0 = vecipa[3]; // nearest distance to IP in z direction
452  double Rvphi0 = vecipa[1];
453 
454  // * WRITE primary vertex position info ("vxyz" branch) *
455  m_vx0 = x0; // primary x-vertex as determined by MDC
456  m_vy0 = y0; // primary y-vertex as determined by MDC
457  m_vz0 = z0; // primary z-vertex as determined by MDC
458  m_vr0 = Rxy; // distance from origin in xy-plane
459  m_rvxy0 = Rvxy0; // nearest distance to IP in xy plane
460  m_rvz0 = Rvz0; // nearest distance to IP in z direction
461  m_rvphi0 = Rvphi0; // angle in the xy-plane (?)
462  m_tuple_vxyz->write(); // "vxyz" branch
463 
464  // * Apply vertex cuts *
465  if(fabs(z0) >= m_vz0cut) continue;
466  if(fabs(Rxy) >= m_vr0cut) continue;
467  if(fabs(Rvz0) >= m_rvz0cut) continue;
468  if(fabs(Rvxy0) >= m_rvxy0cut) continue;
469 
470  // * Add charged track to vector *
471  iGood.push_back(i);
472  nCharge += mdcTrk->charge();
473 
474  }
475
