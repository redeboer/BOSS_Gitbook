361  // * Log number of events *
362  log << MSG::DEBUG << "ncharg, nneu, tottks = "

367

376  log << MSG::DEBUG << "Trigger conditions: " << endmsg;

379  }

383  int m_trig_which(-1);
384  if(m_eventrate) {
385  for(int j=0; j<16; j++) {
386  if(trigData->getTrigChannel(j)) {
387  ++m_trig_tot;
388  m_trig_which = j+1;
389  }

393  }
394  }
395 
396  ++m_cout_col;
397  if(
398  evtRecEvent->totalCharged()<3 ||
399  evtRecTrkCol->size()<3 ||
400  evtRecEvent->totalTracks()>99 ||
401  evtRecTrkCol->size()>99) return StatusCode::SUCCESS;

407  Vint iGood;
408  // Number of different particles: \f$\pi^+\f$, \f$\pi^-\f$, \f$l^+\f$, \f$l^-\f$
409  int m_num[4] = {0, 0, 0, 0};
410  int nCharge = 0;
411  m_pion_matched = 0;
412  m_lep_matched = 0;
413  HepLorentzVector m_lv_pionp;

424  for(int i = 0; i < evtRecEvent->totalCharged(); ++i) {
426  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
427  if(!(*itTrk)->isMdcKalTrackValid()) continue;
428  RecMdcKalTrack* mdcTrk = (*itTrk)->mdcKalTrack();
429 
431  m_vx0 = mdcTrk->x();
432  m_vy0 = mdcTrk->y();
433  m_vz0 = mdcTrk->z();
434  m_vr0 = mdcTrk->r();
435 
437  if(fabs(m_vz0) >= m_vz0cut) continue;
438  if(m_vr0 >= m_vr0cut) continue;
439 
441  iGood.push_back(i);
442  nCharge += mdcTrk->charge();
443 
445  if((*itTrk)->isEmcShowerValid()) {
446  if(mdcTrk->p()<1.) ++m_pion_matched;
447  else ++m_lep_matched;
448  }
449 
451  if(mdcTrk->charge() > 0) {
452  if(mdcTrk->p() < 1.0) { // pi+
453  mdcTrk->setPidType(RecMdcKalTrack::pion);
454  // @warning If you do not modify the `DstMdcKalTrack` package, the `RecMdcKalTrack::p4()` function cannot be used. You should get the momentum from `MdcKalTrack`, then calculate the energy by yourself.
455  m_lv_pionp = mdcTrk->p4(xmass[2]);
456  ++m_num[0];
457  } else { // l+
458  mdcTrk->setPidType(RecMdcKalTrack::electron);
459  m_lv_pos = mdcTrk->p4(xmass[0]);
460  mdcTrk->setPidType(RecMdcKalTrack::muon);
461  m_lv_mup = mdcTrk->p4(xmass[1]);
462  ++m_num[2];
463  }
464  } else {
465  if(mdcTrk->p() < 1.0) { // pi-
466  mdcTrk->setPidType(RecMdcKalTrack::pion);
467  m_lv_pionm = mdcTrk->p4(xmass[2]);
468  ++m_num[1];
469  } else { // l-
470  mdcTrk->setPidType(RecMdcKalTrack::electron);
471  m_lv_ele = mdcTrk->p4(xmass[0]);
472  mdcTrk->setPidType(RecMdcKalTrack::muon);
473  m_lv_mum = mdcTrk->p4(xmass[1]);
474  ++m_num[3];
475  }
476  }
477  }
479 
483  int nGood = iGood.size();
484  log << MSG::DEBUG << "With KalmanTrack, ngood, totcharge = " << nGood << " , " << nCharge << endmsg;
485  if(nGood<3 || nGood>4) return sc;
486  ++m_cout_nGood;
487 
489  m_ep_ratio = 0;
490  for(int i=0; i< evtRecEvent->totalTracks(); ++i) {
491  EvtRecTrackIterator itTrk=evtRecTrkCol->begin() + i;
492  if(!(*itTrk)->isEmcShowerValid()) continue;
493  RecEmcShower *emcTrk = (*itTrk)->emcShower();
494  m_ep_ratio += emcTrk->energy();
495  }
496 
498  if(m_ep_ratio > m_distin_emuon) { // if electron
499  m_lv_lepp = m_lv_pos;
500  m_lv_lepm = m_lv_ele;
501  } else { // if muon
502  m_lv_lepp = m_lv_mup;
503  m_lv_lepm = m_lv_mum;
504  }
505 
507  HepLorentzVector m_lv_lab(0.04, 0, 0, 3.686);
508  if(nGood==4) {
509  if(nCharge) return sc;
510  m_event_flag = 4;
511  } else {
512  if(m_num[0]>1 || m_num[1]>1 || m_num[2]>1 || m_num[3]>1) return sc;
513  if(m_num[0]==0) {
514  if(nCharge != -1) return StatusCode::SUCCESS;
515  m_lv_pionp = m_lv_lab - m_lv_pionm - m_lv_lepp - m_lv_lepm;
516  if(m_lv_pionp.vect().cosTheta()>m_cosThetaCut) return StatusCode::SUCCESS;
517  m_event_flag = 0;
518  }
519  if(m_num[1]==0) {
520  if(nCharge != 1) return StatusCode::SUCCESS;
521  m_lv_pionm = m_lv_lab - m_lv_pionp - m_lv_lepp - m_lv_lepm;
522  if(m_lv_pionm.vect().cosTheta()>m_cosThetaCut) return StatusCode::SUCCESS;
523  m_event_flag = 1;
524  }
525  if(m_num[2]==0) {
526  if(nCharge != -1) return StatusCode::SUCCESS;
527  m_lv_lepp = m_lv_lab - m_lv_pionp - m_lv_pionm - m_lv_lepm;
528  if(m_lv_lepp.vect().cosTheta()>m_cosThetaCut) return StatusCode::SUCCESS;
529  m_event_flag = 2;
530  }
531  if(m_num[3]==0) {
532  if(nCharge != 1) return StatusCode::SUCCESS;
533  m_lv_lepm = m_lv_lab - m_lv_pionp - m_lv_pionm - m_lv_lepp;
534  if(m_lv_lepm.vect().cosTheta()>m_cosThetaCut) return StatusCode::SUCCESS;
535  m_event_flag = 3;
536  }
537  }
538  ++m_cout_mom;
541 
543  HepLorentzVector m_lv_recoil;
544  HepLorentzVector m_lv_jpsi;
545  m_lv_recoil = m_lv_lab - m_lv_pionp - m_lv_pionm;
