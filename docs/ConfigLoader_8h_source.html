146  int ConfigLoader::LoadConfiguration(const char* filename)
147  {
149  std::ifstream file(filename);
150  if(!file.is_open()) {
151  std::cout << "WARNING: Could not load configuration file \"" << filename << "\"" << std::endl;
152  return -1;
153  }
155  std::cout << std::endl << "LOADING CONFIGURATION FROM \"" << filename << "\"" << std::endl;
157  std::string line;
158  while(getline(file, line)) {
161  if(line.back()<' ') line.pop_back();
163  if(line.find('=') == std::string::npos && line.find(' ') == std::string::npos) continue;
165  RemoveLeading(line);
166  RemoveLeading(line, '\t');
168  std::string beginning = line.substr(0,2);
169  if(!beginning.compare("//")) continue; // C++ line comment
170  if(!beginning.compare("/*")) continue; // C++ begin block comment
171  if(beginning.front() == '*') continue; // C++ continue block comment
172  if(beginning.front() == '#') continue; // bash comment
173  if(beginning.front() == '%') continue; // LaTeX comment
175  std::string parname { GetParameterName (line) };
176  std::string parvalue { GetParameterValue(line) };
178  ArgPair_base::SetParameters(parname, parvalue);
180  }
182  ArgPair_base::PrintAll();
184  return ArgPair_base::instances.size();
185  }
