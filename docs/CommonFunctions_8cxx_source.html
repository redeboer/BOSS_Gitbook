188  void CommonFunctions::Fit::FitBWGaussianConvolution(TH1F *hist, const ReconstructedParticle& particle, const UChar_t numPolynomials, TString logScale)
189  {
190 
191  // * Create RooFit variable and data distribution * //
192  RooRealVar fRooRealVar = CreateRooFitInvMassVar(particle);
193  RooDataHist fRooDataHist = CreateRooFitInvMassDistr(hist, fRooRealVar, particle);
194 
195  // * Create double Gaussian function * //
196  RooRealVar m0("GaussianMeanZero", "GaussianMeanZero", 0.);
197  RooRealVar fSigma("#sigma_{gauss}", Form("%s Gaussian width", particle.GetNameLaTeX()),
198  particle.GetSingleGaussianWidth(),
199  Settings::Fit::gSigmaScaleFactorLow * particle.GetSingleGaussianWidth(),
200  Settings::Fit::gSigmaScaleFactorUp * particle.GetSingleGaussianWidth());
201  RooGaussian fGaussian("gauss",
202  Form("Gaussian PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
203  fRooRealVar, m0, fSigma);
204 
205  // * Create Breit-Wigner/Cauchy distribution * //
206  RooRealVar mean(
207  Form("m_{%s}", particle.GetNameLaTeX()),
208  Form("%s mass", particle.GetNameLaTeX()),
209  particle.GetMass(), particle.GetLowerMass(), particle.GetUpperMass());
210  RooRealVar width("#sigma_{BW}", Form("%s BW width", particle.GetNameLaTeX()),
211  particle.GetBWConvolutedWidth(), 0., 10.*particle.GetBWConvolutedWidth());
212  RooBreitWigner bw("breitwigner",
213  Form("Breit-Wigner PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
214  fRooRealVar, mean, width);
215 
216  // * Convolute * //
217  RooFFTConvPdf signal("convolution", "convolution", fRooRealVar, bw, fGaussian);
218  RooRealVar n("N_{gaus1}", "N_{gaus1}", 1e2, 0., 1e6);
219  RooArgList fComponents(signal);
220  RooArgList fNContributions(n);
221 
222  // * Add polynomial background if required * //
223  RooArgList fBckParameters;
224  for(UChar_t i = 0; i <= numPolynomials; ++i) {
225  auto p = new RooRealVar(Form("p%u", i), Form("p%u", i), 0., -1e6, 1e6);
226  fBckParameters.add(*p);
227  }
228  RooChebychev fPolBackground("polBkg",
229  Form("Polynomial-%u background", numPolynomials),
230  fRooRealVar, fBckParameters);
231  RooRealVar fSigToBckRatio(
232  Form("N_{pol%u}", numPolynomials),
233  Form("N_{pol%u}", numPolynomials),
234  0., 0., 1e5);
235  if(numPolynomials) {
236  fComponents.add(fPolBackground);
237  fNContributions.add(fSigToBckRatio);
238  }
239 
240  // * Add the components and fit * //
241  RooAddPdf fFullShape("full_shape", "Double gaussian + background", fComponents, fNContributions);
242  fFullShape.fitTo(
243  fRooDataHist,
244  RooFit::Range(particle.FitFrom(), particle.FitUntil()));
245 
246  // * Plot results and save * //
247  RooPlot *frame = fRooRealVar.frame(); // create a frame to draw
248  frame->SetAxisRange(particle.PlotFrom(), particle.PlotUntil());
249  fRooDataHist.plotOn(frame, // draw distribution
250  RooFit::LineWidth(2), RooFit::LineColor(kBlue+2), RooFit::LineWidth(1),
251  RooFit::MarkerColor(kBlue+2), RooFit::MarkerSize(.5));
252  fFullShape.plotOn(frame, RooFit::LineWidth(2), RooFit::LineColor(kBlack));
253  if(numPolynomials) {
254  fFullShape.plotOn(frame, RooFit::Components(signal), // draw signal
255  RooFit::LineWidth(1), RooFit::LineColor(kRed-4));
256  fFullShape.plotOn(frame, RooFit::Components(fPolBackground), // draw background
257  RooFit::LineStyle(kDashed), RooFit::LineWidth(1), RooFit::LineColor(kGray));
258  }
259  fFullShape.paramOn(frame, RooFit::Layout(.56, .98, .92));
260 
261  // * Write fitted histograms * //
262  TString pname = particle.GetName();
263  pname.ReplaceAll("/", ""); // in case of for instance "J/psi"
264  CommonFunctions::Draw::DrawAndSave(Form("ConvBWSingleGauss_%s", pname.Data()), "", logScale, frame);
265  delete frame;
266 
267  }
