206  void CommonFunctions::Fit::FitBWGaussianConvolution(TH1F *hist, const ReconstructedParticle& particle, const UChar_t numPolynomials, TString logScale)
207  {
208 
209  // * Create RooFit variable and data distribution * //
210  RooRealVar fRooRealVar = CreateRooFitInvMassVar(particle);
211  RooDataHist fRooDataHist = CreateRooFitInvMassDistr(hist, fRooRealVar, particle);
212 
213  // * Create double Gaussian function * //
214  RooRealVar m0("GaussianMeanZero", "GaussianMeanZero", 0.);
215  RooRealVar fSigma("#sigma_{gauss}", Form("%s Gaussian width", particle.GetNameLaTeX()),
216  particle.GetSingleGaussianWidth(),
217  Settings::Fit::gSigmaScaleFactorLow * particle.GetSingleGaussianWidth(),
218  Settings::Fit::gSigmaScaleFactorUp * particle.GetSingleGaussianWidth());
219  RooGaussian fGaussian("gauss",
220  Form("Gaussian PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
221  fRooRealVar, m0, fSigma);
222 
223  // * Create Breit-Wigner/Cauchy distribution * //
224  RooRealVar mean(
225  Form("m_{%s}", particle.GetNameLaTeX()),
226  Form("%s mass", particle.GetNameLaTeX()),
227  particle.GetMass(), particle.GetLowerMass(), particle.GetUpperMass());
228  RooRealVar width("#sigma_{BW}", Form("%s BW width", particle.GetNameLaTeX()),
229  particle.GetBWConvolutedWidth(), 0., 10.*particle.GetBWConvolutedWidth());
230  RooBreitWigner bw("breitwigner",
231  Form("Breit-Wigner PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
232  fRooRealVar, mean, width);
233 
234  // * Convolute * //
235  RooFFTConvPdf signal("convolution", "convolution", fRooRealVar, bw, fGaussian);
236  RooRealVar n("N_{gaus1}", "N_{gaus1}", 1e2, 0., 1e6);
237  RooArgList fComponents(signal);
238  RooArgList fNContributions(n);
239 
240  // * Add polynomial background if required * //
241  RooArgList fBckParameters;
242  for(UChar_t i = 0; i <= numPolynomials; ++i) {
243  auto p = new RooRealVar(Form("p%u", i), Form("p%u", i), 0., -1e6, 1e6);
244  fBckParameters.add(*p);
245  }
246  RooChebychev fPolBackground("polBkg",
247  Form("Polynomial-%u background", numPolynomials),
248  fRooRealVar, fBckParameters);
249  RooRealVar fSigToBckRatio(
250  Form("N_{pol%u}", numPolynomials),
251  Form("N_{pol%u}", numPolynomials),
252  0., 0., 1e5);
253  if(numPolynomials) {
254  fComponents.add(fPolBackground);
255  fNContributions.add(fSigToBckRatio);
256  }
257 
258  // * Add the components and fit * //
259  RooAddPdf fFullShape("full_shape", "Double gaussian + background", fComponents, fNContributions);
260  fFullShape.fitTo(
261  fRooDataHist,
262  RooFit::Range(particle.FitFrom(), particle.FitUntil()));
263 
264  // * Plot results and save * //
265  RooPlot *frame = fRooRealVar.frame(); // create a frame to draw
266  frame->SetAxisRange(particle.PlotFrom(), particle.PlotUntil());
267  fRooDataHist.plotOn(frame, // draw distribution
268  RooFit::LineWidth(2), RooFit::LineColor(kBlue+2), RooFit::LineWidth(1),
269  RooFit::MarkerColor(kBlue+2), RooFit::MarkerSize(.5));
270  fFullShape.plotOn(frame, RooFit::LineWidth(2), RooFit::LineColor(kBlack));
271  if(numPolynomials) {
272  fFullShape.plotOn(frame, RooFit::Components(signal), // draw signal
273  RooFit::LineWidth(1), RooFit::LineColor(kRed-4));
274  fFullShape.plotOn(frame, RooFit::Components(fPolBackground), // draw background
275  RooFit::LineStyle(kDashed), RooFit::LineWidth(1), RooFit::LineColor(kGray));
276  }
277  fFullShape.paramOn(frame, RooFit::Layout(.56, .98, .92));
278 
279  // * Write fitted histograms * //
280  TString pname = particle.GetName();
281  pname.ReplaceAll("/", ""); // in case of for instance "J/psi"
282  CommonFunctions::Draw::DrawAndSave(Form("ConvBWSingleGauss_%s", pname.Data()), "", logScale, frame);
283  delete frame;
284 
285  }
