218  void CommonFunctions::Fit::FitBWGaussianConvolution(TH1F *hist, const ReconstructedParticle& particle, const UChar_t numPolynomials, TString logScale)
219  {
220 
221  // * Create RooFit variable and data distribution * //
222  RooRealVar fRooRealVar = CreateRooFitInvMassVar(particle);
223  RooDataHist fRooDataHist = CreateRooFitInvMassDistr(hist, fRooRealVar, particle);
224 
225  // * Create double Gaussian function * //
226  RooRealVar m0("GaussianMeanZero", "GaussianMeanZero", 0.);
227  RooRealVar fSigma("#sigma_{gauss}", Form("%s Gaussian width", particle.GetNameLaTeX()),
228  particle.GetSingleGaussianWidth(),
229  Settings::Fit::gSigmaScaleFactorLow * particle.GetSingleGaussianWidth(),
230  Settings::Fit::gSigmaScaleFactorUp * particle.GetSingleGaussianWidth());
231  RooGaussian fGaussian("gauss",
232  Form("Gaussian PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
233  fRooRealVar, m0, fSigma);
234 
235  // * Create Breit-Wigner/Cauchy distribution * //
236  RooRealVar mean(
237  Form("m_{%s}", particle.GetNameLaTeX()),
238  Form("%s mass", particle.GetNameLaTeX()),
239  particle.GetMass(), particle.GetLowerMass(), particle.GetUpperMass());
240  RooRealVar width("#sigma_{BW}", Form("%s BW width", particle.GetNameLaTeX()),
241  particle.GetBWConvolutedWidth(), 0., 10.*particle.GetBWConvolutedWidth());
242  RooBreitWigner bw("breitwigner",
243  Form("Breit-Wigner PDF for #it{M}_{%s} distribution", particle.GetDaughterLabel()),
244  fRooRealVar, mean, width);
245 
246  // * Convolute * //
247  RooFFTConvPdf signal("convolution", "convolution", fRooRealVar, bw, fGaussian);
248  RooRealVar n("N_{gaus1}", "N_{gaus1}", 1e2, 0., 1e6);
249  RooArgList fComponents(signal);
250  RooArgList fNContributions(n);
251 
252  // * Add polynomial background if required * //
253  RooArgList fBckParameters;
254  for(UChar_t i = 0; i <= numPolynomials; ++i) {
255  auto p = new RooRealVar(Form("p%u", i), Form("p%u", i), 0., -1e6, 1e6);
256  fBckParameters.add(*p);
257  }
258  RooChebychev fPolBackground("polBkg",
259  Form("Polynomial-%u background", numPolynomials),
260  fRooRealVar, fBckParameters);
261  RooRealVar fSigToBckRatio(
262  Form("N_{pol%u}", numPolynomials),
263  Form("N_{pol%u}", numPolynomials),
264  0., 0., 1e5);
265  if(numPolynomials) {
266  fComponents.add(fPolBackground);
267  fNContributions.add(fSigToBckRatio);
268  }
269 
270  // * Add the components and fit * //
271  RooAddPdf fFullShape("full_shape", "Double gaussian + background", fComponents, fNContributions);
272  fFullShape.fitTo(
273  fRooDataHist,
274  RooFit::Range(particle.FitFrom(), particle.FitUntil()));
275 
276  // * Plot results and save * //
277  RooPlot *frame = fRooRealVar.frame(); // create a frame to draw
278  frame->SetAxisRange(particle.PlotFrom(), particle.PlotUntil());
279  fRooDataHist.plotOn(frame, // draw distribution
280  RooFit::LineWidth(2), RooFit::LineColor(kBlue+2), RooFit::LineWidth(1),
281  RooFit::MarkerColor(kBlue+2), RooFit::MarkerSize(.5));
282  fFullShape.plotOn(frame, RooFit::LineWidth(2), RooFit::LineColor(kBlack));
283  if(numPolynomials) {
284  fFullShape.plotOn(frame, RooFit::Components(signal), // draw signal
285  RooFit::LineWidth(1), RooFit::LineColor(kRed-4));
286  fFullShape.plotOn(frame, RooFit::Components(fPolBackground), // draw background
287  RooFit::LineStyle(kDashed), RooFit::LineWidth(1), RooFit::LineColor(kGray));
288  }
289  fFullShape.paramOn(frame, RooFit::Layout(.56, .98, .92));
290 
291  // * Write fitted histograms * //
292  TString pname = particle.GetName();
293  pname.ReplaceAll("/", ""); // in case of for instance "J/psi"
294  CommonFunctions::Draw::DrawAndSave(Form("ConvBWSingleGauss_%s", pname.Data()), "", logScale, frame);
295  delete frame;
296 
297  }
