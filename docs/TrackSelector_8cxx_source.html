392  void TrackSelector::CreateChargedCollection()
393  {
396  if(!fCreateChargedCollection) return;
397 
399  fGoodChargedTracks.clear();
400  fPIDInstance = ParticleID::instance();
401 
403  if(!fEvtRecEvent->totalCharged())
404 
406  fNChargesMDC = 0;
407  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
408  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
411 
412  // * Get track info from Main Drift Chamber
413  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
414  fTrackIterator = fEvtRecTrkCol->begin() + i;
415  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
416  ++fCounter_Nmdcvalid;
417  fTrackMDC = (*fTrackIterator)->mdcTrack();
418 
419  // * Get kinematics of track
420  double phi = fTrackMDC->helix(1);
421  double vr =
422  (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
423  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
424 
425  // * Get radii of track vertex
426  HepVector a = fTrackMDC->helix();
427  HepSymMatrix Ea = fTrackMDC->err();
428  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
429  VFHelix helixip(point0, a, Ea);
430  helixip.pivot(fVertexPoint);
431  HepVector vecipa = helixip.a();
432  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
433  double rvz = vecipa[3]; // nearest distance to IP in z direction
434  double rvphi = vecipa[1]; // angle in the xy-plane (?)
435 
438  if(fCut_Vz .FailsMax(fabs(fTrackMDC->z()))) continue;
439  if(fCut_Vxy.FailsMax(vr) ) continue;
440  if(fCut_Rz .FailsMax(fabs(rvz)) ) continue;
441  if(fCut_Rxy.FailsMax(rvxy) ) continue;
442  if(fCut_CosTheta.FailsMax(fabs(cos(fTrackMDC->theta())))) continue;
443 
445  // * Add charged track to vector
446  fGoodChargedTracks.push_back(*fTrackIterator);
447  fNChargesMDC += fTrackMDC->charge(); // @todo Check if this makes sense at all
448 
450  if(fNTuple_charged.DoWrite()) {
451  fNTuple_charged.at("vx") = fTrackMDC->x();
452  fNTuple_charged.at("vy") = fTrackMDC->y();
453  fNTuple_charged.at("vz") = fTrackMDC->z();
454  fNTuple_charged.at("vr") = vr;
455  fNTuple_charged.at("rvxy") = rvxy;
456  fNTuple_charged.at("rvz") = rvz;
457  fNTuple_charged.at("rvphi") = rvphi;
458  fNTuple_charged.at("phi") = phi;
459  fNTuple_charged.at("p") = fTrackMDC->p();
460  fNTuple_charged.Write();
461  }
462 
464  WriteDedxInfo(*fTrackIterator, fNTuple_dedx);
465 
467  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite()) {
468  // * Check if MDC and TOF info for track are valid *
469  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
470  if(!(*fTrackIterator)->isTofTrackValid()) continue;
471  // * Get momentum as determined by MDC *
472  fTrackMDC = (*fTrackIterator)->mdcTrack();
473  double ptrk;
474  if(fTrackMDC) ptrk = fTrackMDC->p();
475  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
476  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
477  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
478  TofHitStatus hitStatus;
479  hitStatus.setStatus((*iter_tof)->status());
480  if(!hitStatus.is_counter()) continue;
481  if(hitStatus.is_barrel()) {
482  if(hitStatus.layer() == 1) { // inner barrel
483  WriteTofInformation(iter_tof, ptrk, fNTuple_TofIB);
484  } else if(hitStatus.layer() == 2) { // outer barrel
485  WriteTofInformation(iter_tof, ptrk, fNTuple_TofOB);
486  }
487  }
488  else if(hitStatus.layer() == 0) // end cap
489  WriteTofInformation(iter_tof, ptrk, fNTuple_TofEC);
490  }
491  }
492 
494  }
495  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fGoodChargedTracks.size() << endmsg;
496 
498  }

554  void TrackSelector::CreateMCtruthCollection()
555  {
558  if(!fNTuple_mctruth.DoWrite()) return;
559 
561  if(fEventHeader->runNumber()>=0) return;
562 
564  fMcParticleCol = SmartDataPtr<Event::McParticleCol>(eventSvc(), "/Event/MC/McParticleCol");
565 
567  if(!fMcParticleCol) {
568  fLog << MSG::ERROR << "Could not retrieve McParticelCol" << endmsg;
569  return;
570  }
571 
573  fNTuple_mctruth.runID = fEventHeader->runNumber();
574  fNTuple_mctruth.evtID = fEventHeader->eventNumber();
575 
577  bool doNotContinue(true); // only start recording if set to false in the loop
578  int rootIndex(-1); // index of mother particles, necessary for topoana
579  fNTuple_mctruth.index = 0;
580 
582  for(Event::McParticleCol::iterator it = fMcParticleCol->begin(); it!=fMcParticleCol->end(); ++it) {
586  if( (*it)->primaryParticle()) continue;
587  if(!(*it)->decayFromGenerator()) continue;
589  if(NTupleTopoAna::IsInitialCluster(*it)) {
590  doNotContinue = false;
591  rootIndex = (*it)->trackIndex();
592  }
594  if(doNotContinue) continue;
596  fNTuple_mctruth.particle[fNTuple_mctruth.index] = (*it)->particleProperty();
598  if(NTupleTopoAna::IsInitialCluster(*it))
599  fNTuple_mctruth.mother[fNTuple_mctruth.index] = -1;
600  else
601  fNTuple_mctruth.mother[fNTuple_mctruth.index] = ((*it)->mother()).trackIndex()-rootIndex-1;
603  fNTuple_mctruth.index++;
605  fMcParticles.push_back(*it);
607  }

742  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, NTupleContainer &tuple)
743  {
745  if(!tuple.DoWrite()) return;
746 
748  double path = (*iter_tof)->path();
749  std::vector<double> texp(gNMasses);
750  for(size_t j = 0; j < texp.size(); ++j) {
751  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
752  double beta = gb/sqrt(1+gb*gb);
753  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF
754  }
755 
757  tuple.at("p") = ptrk;
758  tuple.at("tof") = (*iter_tof)->tof();
759  tuple.at("path") = path;
760  tuple.at("cntr") = (*iter_tof)->tofID();
761  tuple.at("ph") = (*iter_tof)->ph();
762  tuple.at("zrhit") = (*iter_tof)->zrhit();
763  tuple.at("qual") = (*iter_tof)->quality();
764  tuple.at("tof_e") = path - texp[0];
765  tuple.at("tof_mu") = path - texp[1];
766  tuple.at("tof_pi") = path - texp[2];
767  tuple.at("tof_K") = path - texp[3];
768  tuple.at("tof_p") = path - texp[4];
