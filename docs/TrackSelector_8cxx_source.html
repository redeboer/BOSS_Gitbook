188  StatusCode TrackSelector::execute()
189  {
190  fLog << MSG::DEBUG << "===>> TrackSelector::execute() <<===" << endmsg;
191 
200 
201  // * Load headers
202  fLog << MSG::DEBUG << "Loading EventHeader, EvtRecEvent, and EvtRecTrackCol" << endmsg;
203  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
204  fEvtRecEvent = SmartDataPtr<EvtRecEvent> (eventSvc(), EventModel::EvtRec::EvtRecEvent);
205  fEvtRecTrkCol = SmartDataPtr<EvtRecTrackCol> (eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
206 
207  // * Log run number, event number, and number of events *
208  fLog << MSG::DEBUG
209  << "RUN " << fEventHeader->runNumber() << ", "
210  << "event number " << fEventHeader->eventNumber() << endmsg;
211  fLog << MSG::DEBUG
212  << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "
213  << "Nneutral = " << fEvtRecEvent->totalNeutral() << ", "
214  << "Ntotal = " << fEvtRecEvent->totalTracks() << endmsg;
215 
216  // * Increase counters *
217  ++fCounter_Nevents;
218  fCounter_Ncharged += fEvtRecEvent->totalCharged();
219  fCounter_Nneutral += fEvtRecEvent->totalNeutral();
220  fCounter_Ntracks += fEvtRecEvent->totalTracks();
221 
222  // * Set vertex origin *
223  IVertexDbSvc* vtxsvc;
224  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
225  if(vtxsvc->isVertexValid()){
226  double* dbv = vtxsvc->PrimaryVertex();
227  // double* vv = vtxsvc->SigmaPrimaryVertex();
228  fVertexPoint.set(dbv[0], dbv[1], dbv[2]);
229  }
230 
232  if(fEventHeader->runNumber()<0 && fNTuple_mctruth) {
235  fMcParticleCol = SmartDataPtr<Event::McParticleCol>(eventSvc(), "/Event/MC/McParticleCol");
236 
239  if(!fMcParticleCol) {
240  fLog << MSG::ERROR << "Could not retrieve McParticelCol" << endmsg;
241  return(StatusCode::FAILURE);
242  }
243 
245  bool doNotRecord(true);
246  for(Event::McParticleCol::iterator it = fMcParticleCol->begin(); it!=fMcParticleCol->end(); ++it) {
249  if( (*it)->primaryParticle()) continue;
251  if(!(*it)->decayFromGenerator()) continue;
253  if(
254  (*it)->particleProperty() == 91 ||
255  (*it)->particleProperty() == 92 ) {
256  doNotRecord = false;
257  }
259  if(doNotRecord) continue;
261  fMcParticles.push_back(*it);
263  } // end of for loop
265 
266  } // end of if runNumber<0
267 
270 
271  // * Reset counters and initialize PID instance *
272  fGoodChargedTracks.clear();
273  fPIDInstance = ParticleID::instance();
274 
275  // * Only perform if there are charged tracks *
276  if(fNTuple_charged && fEvtRecEvent->totalCharged()) {
277 
278  fNChargesMDC = 0;
279  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
280  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
283 
284  // * Get track info from Main Drift Chamber
285  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
286  fTrackIterator = fEvtRecTrkCol->begin() + i;
287  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
288  ++fCounter_Nmdcvalid;
289  fTrackMDC = (*fTrackIterator)->mdcTrack();
290 
291  // * Get kinematics of track
292  double phi = fTrackMDC->helix(1);
293  double vr =
294  (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
295  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
296 
297  // * Get radii of track vertex
298  HepVector a = fTrackMDC->helix();
299  HepSymMatrix Ea = fTrackMDC->err();
300  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
301  VFHelix helixip(point0, a, Ea);
302  helixip.pivot(fVertexPoint);
303  HepVector vecipa = helixip.a();
304  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
305  double rvz = vecipa[3]; // nearest distance to IP in z direction
306  double rvphi = vecipa[1]; // angle in the xy-plane (?)
307 
310  if(fCut_Vz .FailsMax(fabs(fTrackMDC->z()))) continue;
311  if(fCut_Vxy.FailsMax(vr) ) continue;
312  if(fCut_Rz .FailsMax(fabs(rvz)) ) continue;
313  if(fCut_Rxy.FailsMax(rvxy) ) continue;
314  if(fCut_CosTheta.FailsMax(fabs(cos(fTrackMDC->theta())))) continue;
315 
317  // * Add charged track to vector
318  fGoodChargedTracks.push_back(*fTrackIterator);
319  fNChargesMDC += fTrackMDC->charge(); // @todo Check if this makes sense at all
320 
322  if(fNTuple_charged.DoWrite()) {
323  fNTuple_charged.at("vx") = fTrackMDC->x();
324  fNTuple_charged.at("vy") = fTrackMDC->y();
325  fNTuple_charged.at("vz") = fTrackMDC->z();
326  fNTuple_charged.at("vr") = vr;
327  fNTuple_charged.at("rvxy") = rvxy;
328  fNTuple_charged.at("rvz") = rvz;
329  fNTuple_charged.at("rvphi") = rvphi;
330  fNTuple_charged.at("phi") = phi;
331  fNTuple_charged.at("p") = fTrackMDC->p();
332  fNTuple_charged.Write();
333  }
334 
336  WriteDedxInfo(*fTrackIterator, fNTuple_dedx);
337 
339  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite()) {
340 
341  // * Check if MDC and TOF info for track are valid *
342  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
343  if(!(*fTrackIterator)->isTofTrackValid()) continue;
344 
345  // * Get momentum as determined by MDC *
346  fTrackMDC = (*fTrackIterator)->mdcTrack();
347  double ptrk;
348  if(fTrackMDC) ptrk = fTrackMDC->p();
349  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
350  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
351  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
352  TofHitStatus hitStatus;
353  hitStatus.setStatus((*iter_tof)->status());
354  if(!hitStatus.is_counter()) continue;
355  if(hitStatus.is_barrel()) {
356  if(hitStatus.layer() == 1) { // inner barrel
357  if(fNTuple_TofIB.DoWrite()) WriteTofInformation(iter_tof, ptrk, fNTuple_TofIB);
358  } else if(hitStatus.layer() == 2) { // outer barrel
359  if(fNTuple_TofOB.DoWrite()) WriteTofInformation(iter_tof, ptrk, fNTuple_TofOB);
360  }
361  }
362  else if(fNTuple_TofEC.DoWrite() && hitStatus.layer() == 0) // end cap
363  WriteTofInformation(iter_tof, ptrk, fNTuple_TofEC);
364  }
365 
366  } // if(fNTuple_tofec.DoWrite() || fNTuple_tofib.DoWrite() || fNTuple_tofob.DoWrite())
367 
369  }
370 
371  // * Finish good charged track selection *
372  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fGoodChargedTracks.size() << endmsg;
373  }
374 
375 
378 
379  // * Set counters and data members *
380  fGoodNeutralTracks.clear();
381 
382  // * Only perform if there are neutral tracks *
383  if(fNTuple_neutral && fEvtRecEvent->totalNeutral()) {
384 
385  // * Loop over neutral tracks *
386  fLog << MSG::DEBUG << "Starting 'good' neutral track selection:" << endmsg;
387  for(int i = fEvtRecEvent->totalCharged(); i < fEvtRecEvent->totalTracks(); ++i) {
390  fLog << MSG::DEBUG << " neutral track " << i-fEvtRecEvent->totalCharged() << "/" << fEvtRecEvent->totalNeutral() << endmsg;
391  fTrackIterator = fEvtRecTrkCol->begin() + i;
392  if(!(*fTrackIterator)->isEmcShowerValid()) continue;
393  fTrackEMC = (*fTrackIterator)->emcShower();
394  if(!fTrackEMC) continue;
395 
397  if(fCut_PhotonEnergy.FailsMin(fTrackEMC->energy())) continue;
398 
400  if(fNTuple_neutral.DoWrite()) {
401  fNTuple_neutral.at("E") = fTrackEMC->energy();
402  fNTuple_neutral.at("x") = fTrackEMC->x();
403  fNTuple_neutral.at("y") = fTrackEMC->y();
404  fNTuple_neutral.at("z") = fTrackEMC->z();
405  fNTuple_neutral.at("phi") = fTrackEMC->phi();
406  fNTuple_neutral.at("theta") = fTrackEMC->theta();
407  fNTuple_neutral.at("time") = fTrackEMC->time();
408  NTupleContainer::Get("neutral").Write();
409  }
410 
412  fGoodNeutralTracks.push_back(*fTrackIterator);
413 
415  }
416 
417  // * Finish good photon selection *
418  fLog << MSG::DEBUG << "Number of 'good' photons: " << fGoodNeutralTracks.size() << endmsg;
419  }
420 
421 
423  if(fNTuple_mult.DoWrite()) {
424  fNTuple_mult.at("Ntotal") = fEvtRecEvent->totalTracks();
425  fNTuple_mult.at("Ncharge") = fEvtRecEvent->totalCharged();
426  fNTuple_mult.at("Nneutral") = fEvtRecEvent->totalNeutral();
427  if(fNTuple_charged) {
428  fNTuple_mult.at("NgoodCharged") = fGoodChargedTracks.size();
429  fNTuple_mult.at("Nmdc") = fNChargesMDC;
430  }
431  if(fNTuple_neutral) fNTuple_mult.at("NgoodNeutral") = fGoodNeutralTracks.size();
432  NTupleContainer::Get("mult").Write();
433  }
434 
436  if(fNTuple_vertex.DoWrite()) {
437  fNTuple_vertex.at("vx0") = fVertexPoint.x();
438  fNTuple_vertex.at("vy0") = fVertexPoint.y();
439  fNTuple_vertex.at("vz0") = fVertexPoint.z();
440  NTupleContainer::Get("vertex").Write();
441  }
442 
444  if(execute_rest() == StatusCode::SUCCESS) return StatusCode::SUCCESS;
445  if(fPIDInstance) delete fPIDInstance;
446 
448  return StatusCode::SUCCESS;
449  }

683  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, NTupleContainer &tuple)
684  {

689  double path = (*iter_tof)->path();
690  std::vector<double> texp(gNMasses);
691  for(size_t j = 0; j < texp.size(); ++j) {
692  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
693  double beta = gb/sqrt(1+gb*gb);
694  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF
