25  BOSSOutputLoader::BOSSOutputLoader(const char* path, bool print_branches, bool print_averages) :
26  fDirectoryPath(path)
27  {
30  if(fDirectoryPath.EndsWith(".root")) {
33  TFile file(fDirectoryPath.Data());
34  if(file.IsZombie()) return;
36  TString fname(fDirectoryPath);
37  auto pos = fDirectoryPath.Last('/');
38  fDirectoryPath.Resize(pos);
39  fname.Remove(0, pos+1);
41  fFileNames.push_back(fname);
43  }
45  else {
48  if(fDirectoryPath.EndsWith("/")) fDirectoryPath.Chop();
50  TSystemDirectory dir(path, path);
52  TList *files = dir.GetListOfFiles();
53  if(!files) {
54  std::cout << "ERROR: Directory \"" << fDirectoryPath << "\" does not exist" << std::endl;
55  return;
56  }
58  TIter fileIter(files);
59  TObject *obj;
60  while ((obj = fileIter())) {
61  auto systemFile = dynamic_cast<TSystemFile*>(obj);
62  if(!systemFile) continue;
63  TString filename { systemFile->GetName() };
64  if(!filename.EndsWith(".root")) continue;
65  TString fname(systemFile->GetName());
67  TFile file((fDirectoryPath+"/"+fname).Data());
68  if(!file.IsZombie()) fFileNames.push_back(fname);
69  }
71  }
73  LoadChains(print_branches, print_averages);
74  }

330  std::list<ChainLoader*> BOSSOutputLoader::CreateOrderedMap()
331  {
332  // * Create list of pointers to `ChainLoader` * //
333  std::list<ChainLoader*> outputList;
334  for(auto it = fChains.begin(); it != fChains.end(); ++it) {
335  TString name(it->second.GetChain().GetName());
336  if(!name.EqualTo("_cutvalues")) outputList.push_back(&(it->second));
337  }
338  // * Sort resulting list based on number of entries * //
339  outputList.sort([](ChainLoader* const & a, ChainLoader* const & b)
340  {
341  std::string nameA(a->GetChain().GetName());
342  std::string nameB(b->GetChain().GetName());
343  return a->GetEntries() != b->GetEntries() ?
344  a->GetEntries() > b->GetEntries() :
345  nameA < nameB;
346  });
347  return outputList;
348  }
